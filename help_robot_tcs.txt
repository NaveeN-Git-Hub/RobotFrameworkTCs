# To Setup Jenkins in Ubuntu  #

-- Install packages
wget -q -O - https://pkg.jenkins.io/debian-stable/jenkins.io.key | sudo apt-key add -
sudo sh -c 'echo deb https://pkg.jenkins.io/debian-stable binary/ > /etc/apt/sources.list.d/jenkins.list'
sudo apt-get update
sudo apt-get install jenkins
sudo dpkg --configure -a (if any error)
sudo apt-get install jenkins
sudo apt search openjdk
sudo apt install openjdk-11-jdk
sudo apt-get update
sudo apt install openjdk-11-jdk
java -version
 
-- Configure Jenkins 
  - Open http://localhost:8080
  - Get the initial password using "sudo cat /var/lib/jenkins/secrets/initialAdminPassword"
  - And Paste in the GUI
  - Install Suggested plugins
  - Change the Port number 8080 TO 9292 in "sudo vim /etc/default/jenkins"
  - Restart the Server using "sudo /etc/init.d/jenkins restart"
  
  - To Start the Jenkins:
     sudo /etc/init.d/jenkins start
	 http://localhost:9292

# Setup and Instal Java and Jenkins in Windows  #

Download 'jdk-11.0.12_windows-x64_bin.exe'
Install and give path in env 
Download jenkins.war
Go to jenkins.war path and execute the below command
java -jar jenkins.war --httpPort=9191
open web using 'http://localhost:9191/'
and setup the password

# In the Jenkins GUI - To Execute commands in Wondows

Create New Freestyle JOB 

-- Source Code Management
   None
   
-- Build Triggers
   Select "Build periodically"
   */5 * * * * means every 5 minutes

-- Execute Windows Batch command
   robot --outputdir my_robot_results ApiAutomation.robot
   exit 0

-- Post-build Actions
   Select "Publish Robot Framework test results"
   Directory of Robot output: my_robot_results
  
Apply and Save


# In the Jenkins GUI - To Execute commands from Git HUb

First push the code into GitHub

git init
git status
git add ApiAutomation.robot
git add help_robot_tcs.txt
git add rf_userdefined_keywords.py
git add json_file.json
git status
git commit -m "Test github integration with jenkins"

Create Repository - "RobotFrameworkTCs" in GitHub

git remote add origin git@github.com:NaveeN-Git-Hub/RobotFrameworkTCs.git
git push -u origin master



Then Create New Freestyle JOB in Jenkins GUI

-- Source Code Management
   Git
   
   -- Repository URL
      https://github.com/NaveeN-Git-Hub/RobotFrameworkTCs.git
   -- Add Credentials
   
-- Branches to build
   */master
   
-- Build Triggers
   Select "Build periodically"
   */5 * * * * means every 5 minutes

-- Build
-- Execute Windows Batch command
   robot --outputdir my_robot_results ApiAutomation.robot
   exit 0

-- Post-build Actions
   Select "Publish Robot Framework test results"
   Directory of Robot output: my_robot_results
  
Apply and Save
